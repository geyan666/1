#include <stdio.h>
#include <math.h>
int main(){
  double x,y;;
  char o; 
    scanf("%lf%c%lf",&x,&o,&y);

  if(o=='+') 
  {
    printf("=%.2lf",x+y) ; 
  }

    else if(o=='-') 
       {
     printf("=%.2lf",x-y) ; 
  }  
    else if(o=='*') 
         {
   printf("=%.2lf",x*y) ; 
  }
    else if(o=='/') 
    {
       if(y==0)
         printf("分母不能为零");   	
        else   	
          printf("=%.2lf",x/y) ; 
  }
    
    else 
  {
                    
printf("未知运算符") ; 
  }

return 0;
}
